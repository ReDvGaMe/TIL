## 3. 코드  
```
#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;

	int k = 1;
	for (int i = 0; i < 2 * n - 1; i++)
	{
		for (int j = 0; j < n - k; j++)
			cout << " ";
		for (int j = 0; j < 1 + ((k - 1) * 2); j++)
			cout << "*";

		cout << endl;
		k = (i < n - 1) ? k + 1 : k - 1;
	}

	return 0;
}
```

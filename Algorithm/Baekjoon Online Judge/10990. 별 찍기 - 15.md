## 3. 코드  
```
#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n - i - 1; j++)
			cout << " ";
		cout << "*";
		for (int j = 0; j < i * 2 - 1; j++)
			cout << " ";
		if(i != 0)
			cout << "*";

		cout << endl;
	}

	return 0;
}
```
